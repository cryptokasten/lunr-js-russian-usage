*** Creating index

#+BEGIN_SRC js
var idx = lunr(function () {
  this.use(lunr.ru)
  this.ref('id')
  this.field('text')

  documents.forEach(function (doc) {
    this.add(doc)
  }, this)
})
#+END_SRC

*** Search

#+BEGIN_SRC js
idx.search("жена")
